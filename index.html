data <- read_csv("data/diversity.csv")

#tidy data
data <- data %>% 
  mutate(rii_calc = (shrub-open)/(shrub + open)) %>%
  filter(!is.na(n)) %>%
  filter(!is.na(rii_calc)) %>%
  mutate(lat = as.numeric(lat), n = as.numeric(n)) %>%
  mutate(var.es = (shrub_error/(sqrt(n)))) %>%
  filter(var.es > 0) %>% 
  filter(!is.na(simplified_response))
